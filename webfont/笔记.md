## new

```javascript
    function Cnew(){
        let newObj = Object.create({});
        let originObj = [].shift.call(arguments);
        if(typeof originObj != 'function'){
            throw 'wrong';
        }
        
        newObj.__proto__ = originObj.prototype;
        let res = originObj.apply(newObj,arguments);
        return (typeof res) == 'object'  ? res:newObj;
    }
```
